# TeXworks: Patterns for syntax coloring
# Each entry consists of three whitespace-separated fields:
# &lt;style&gt; &lt;spell?&gt; &lt;regex&gt;
# &lt;style&gt; is a combination of color and style flags
# Valid syntax:
# &lt;fgcolor&gt;
# &lt;fgcolor&gt;/&lt;bgcolor&gt;
# ;&lt;fontflags&gt;
# &lt;fgcolor&gt;;&lt;fontflags&gt;
# &lt;fgcolor&gt;/&lt;bgcolor&gt;;&lt;fontflags&gt;
#
# The color fields &lt;fgcolor&gt; and &lt;bgcolor&gt; are either:
# *) a color name from the list defined in the SVG standard;
#    see http://www.w3.org/TR/SVG/types.html#ColorKeywords
# *) a hexadecimal color value of the form #rrggbb; note that if this is used at
#    the beginning of a line, a space &quot; &quot; must be added in front of it so the
#    line is not interpreted as a comment (e.g., &quot; #aabbcc&quot; instead of &quot;#aabbcc&quot;)
#
# The &lt;fontflags&gt; field is a combination of the letters
# B (bold)
# I (italic)
# U (underlined)
# &lt;spell&gt; is a flag controlling whether this style is subject to spell-checking;
# &quot;Y&quot; or &quot;y&quot; for yes, &quot;N&quot; (or anything else, actually) for no.
# &lt;regex&gt; is the regular expression to match.
# Note that order of rules may be important if several rules could match at the same position;
# this is why the generic &quot;control sequence&quot; rule comes AFTER the specific \begin/\end one.
[LaTeX]
# special characters
# PURPLE
 #ae8ef0	N	[$#^_{}&amp;] 
# LaTeX environments
# ORANGE
 #dc861f;I	N	\\(?:begin|end)\s*\{[^}]*\} 
# LaTeX packages
# PINK
 #ec4271;I	N	\\usepackage\s*(?:\[[^]]*\]\s*)?\{[^}]*\}
# control sequences
# LIGHTBLUE
 #5dd4ef	N	\\(?:[A-Za-z@]+|.) 
# comments
# DARKBROWN
 #797666;I	Y	%.* 
[ConTeXt]
# special characters
# PURPLE
 #ae8ef0	N	[$#^_{}&amp;] 
# start/stop
# ORANGE
 #dc861f;I	N	\\(?:start|stop)[A-Za-z]+ 
# control sequences
# LIGHTBLUE
 #5dd4ef	N	\\(?:[A-Za-z@]+|.) 
# comments
# DARKBROWN
 #797666;I	Y	%.*
